Код програми:

                window.onload = function() {
                    const square = document.getElementById("square");
                    const shrinkButton = document.getElementById("shrinkButton");
                    const expandButton = document.getElementById("expandButton");
                
                    // Зменшення розміру квадрата
                    shrinkButton.addEventListener("click", function() {
                        let currentSize = parseInt(window.getComputedStyle(square).width); // Отримуємо поточний розмір
                        square.style.width = (currentSize - 15) + "px";  // Зменшуємо ширину
                        square.style.height = (currentSize - 15) + "px"; // Зменшуємо висоту
                    });
                
                    // Збільшення розміру квадрата
                    expandButton.addEventListener("click", function() {
                        let currentSize = parseInt(window.getComputedStyle(square).width); // Отримуємо поточний розмір
                        square.style.width = (currentSize + 15) + "px";  // Збільшуємо ширину
                        square.style.height = (currentSize + 15) + "px"; // Збільшуємо висоту
                    });
                };
