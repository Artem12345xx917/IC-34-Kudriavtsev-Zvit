                // Завдання 6 - подвоєння значень у списку
const doubleButton = document.getElementById('doubleButton');
const numberList = document.getElementById('numberList');

// Обробник події для кнопки "Подвоїти"
doubleButton.addEventListener('click', function() {
    const listItems = numberList.getElementsByTagName('li');  // Отримуємо всі елементи списку

    // Перебираємо кожен елемент списку
    for (let i = 0; i  listItems.length; i++) {
        let currentValue = parseInt(listItems[i].textContent);  // Отримуємо поточне значення
        listItems[i].textContent = currentValue * 2;  // Подвоюємо значення
    }
});
